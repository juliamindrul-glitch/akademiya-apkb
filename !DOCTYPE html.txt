        function initializeTracking() {
            // Cookie tracking implementation
            // Tracks: page views, time on site, clicked products, chat interactions
            const tracking = {
                sessionId: Date.now(),
                startTime: new Date().toISOString(),
                pages: [window.location.pathname],
                clicks: [],
                source: document.referrer || 'direct'
            };
            
            localStorage.setItem('tracking', JSON.stringify(tracking));
            
            // Track clicks
            document.addEventListener('click', (e) => {
                const target = e.target.closest('.card, .btn, a');
                if (target) {
                    const data = JSON.parse(localStorage.getItem('tracking') || '{}');
                    data.clicks.push({
                        element: target.tagName,
                        text: target.textContent?.substring(0, 50),
                        time: new Date().toISOString()
                    });
                    localStorage.setItem('tracking', JSON.stringify(data));
                }
            });
        }

            } else {
                initializeTracking();
